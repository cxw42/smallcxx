78 Matcher::addGlob(const smallcxx::glob::Path& glob,
79  const smallcxx::glob::Path& path)
80 {

83  throw domain_error("Matcher::addGlob: path must be nonempty and end with /");
84  }
85 
86  // Strip trailing slash. TODO handle this in a cleaner way.
87  const auto pathNoSlash = path.substr(0, path.size() - 1);
88 
89  Polarity polarity = (glob[0] == '!') ? Polarity::Exclude : Polarity::Include;
90 
91  string fullGlob; // new glob of (essentially) `path`/`glob`
92 
93  /* fullGlob is:
94  * - path[double_star]/[glob] if glob does not contain '/'
95  * - path[glob] if glob starts with a '/'
96  * - path/[glob] if glob contains '/' but does not start with '/'.
97  *
98  * If the path part has any special characters as defined by ec_glob.c, we
99  * need to escape them.
100  */
101 
102  /* Escaping special characters in the directory part. */
103  size_t lastpos = 0;
104  size_t pos;
105  while((pos = pathNoSlash.find_first_of(ec_special_chars,
106  lastpos)) != pathNoSlash.npos) {
107  fullGlob += pathNoSlash.substr(lastpos, pos - lastpos);
108  fullGlob += '\\'; /* escaping char */
109  fullGlob += pathNoSlash[pos];
110  lastpos = pos + 1;
111  }
112 
113  // Copy any of the string that may be left
114  fullGlob += pathNoSlash.substr(lastpos);
115 
116  if (glob.find('/') == glob.npos) { // No / is found, append '[star][star]/'
117  fullGlob += "**/";
118 
119  } else if (glob[0] != '/') {
120  // The first char is not '/' but section contains '/', append a '/'
121  fullGlob += "/";
122  }
123

126  } else {

128  fullGlob += &glob[1];

160 PathCheckResult
161 Matcher::check(const smallcxx::glob::Path& path) const
162 {
163  if(!ready()) {
164  throw logic_error("Matcher: Call to check() or contains() when not ready --- call finalize() after adding globsets");
165  }
166 
167  if(path.empty()) { // no matcher contains an empty path
168  return PathCheckResult::Unknown;
169  }
170 
171  if(*path.cbegin() != '/') {
172  throw domain_error(
173  "Matcher::contains: path must be absolute (start with /)");
174  }
175 
176  // Check the globsets from back to front because later entries override
177  // earlier entries.
178  for(auto it = globsets_.crbegin(); it != globsets_.crend(); ++it) {
179  if(it->globSet.contains(path)) {
180  return (it->polarity == Polarity::Include) ?
181  PathCheckResult::Included : PathCheckResult::Excluded;
182  }
183  }
