128 {

130  static intmax_t pid = getpid();

132  static const char *endcolor = tty ? NORMAL : "";

136  char msg[LOGBUF_NBYTES];
137  char whole[LOGBUF_NBYTES];
138  int charsWritten;
139 
140  // Accept the possibility of missing a log message around the time
141  // the level changes.
142  if(msgLevel > g_currSystemLevels[domain].l) {
143  return;
144  }

147  clock_gettime(CLOCK_BOOTTIME, &ts);
148  const double boottime = ts.tv_sec + (double)ts.tv_nsec / 1e9;

153  // Uncovered --- I don't know any way to cause this to happen so I can test it
154  preamble[0] = '\0'; //keep going without a timestamp // LCOV_EXCL_LINE
155  }
156

158  if(stimestamp.length() > 16) {
159  stimestamp = stimestamp.substr(stimestamp.length() - 16);
160  }
161 
162  // Level
163  const char *levelname =
164  ((msgLevel < LOG_MIN) || (msgLevel > LOG_MAX)) ? "" : g_levelnames[msgLevel];
165  const char *bodycolor = !tty ? "" : (
166  (msgLevel == LOG_ERROR) ? RED :
167  (msgLevel == LOG_WARNING) ? YELLOW :
168  (msgLevel == LOG_FIXME) ? WHITE :
169  NORMAL
170  );
171

173  charsWritten = snprintf(preamble, sizeof(preamble),
174  "[%16.16s] %s%-8" PRIdMAX "%s %-5.5s %20.20s:%-4d %-20.20s ",
175  stimestamp.c_str(),
176  pidcolor, pid, bodycolor, levelname, file, line, function);
177

179  // *INDENT-OFF*
180  // Uncovered --- I don't know any way to cause this to happen so I can test it
181  const char msg[] = "Dropped log message (preamble error)\n"; // LCOV_EXCL_START

191  // *INDENT-OFF*
192  // Uncovered; same reason as above
193  const char msg[] = "Dropped log message (message error)\n"; // LCOV_EXCL_START

200

203  - 2 - strlen(endcolor); // -2 = room for "\n\0"
204  charsWritten = snprintf(whole, sizeof(whole), "%s%-.*s%s\n", preamble,
205  roomForMessage, msg, endcolor);

304 static bool
305 parseDetail(const char *detailEnvVarName)
306 {
307  char *detail;
308  if(!detailEnvVarName || (detailEnvVarName[0] == '\0') ||
309  ((detail = getenv(detailEnvVarName)) == nullptr)) {
310  return false;
311  }
312 
313  enum { get_domain, get_value } state;
314  state = get_domain;
315  string domain, value;
316  for(const auto c : string(detail)) {
317  if(isspace(c)) {
318  continue;
319  }
320  switch(state) {
321  case get_domain:
322  if(c == ':') {
323  state = get_value;
324  continue;
325  }
326  domain += c;
327  break;
328 
329  case get_value:
330  if(c == ',') {
331  // Finish
332  setLevelFromStrings(domain, value);
333 
334  // Set up for the next one, if any
335  state = get_domain;
336  domain = "";
337  value = "";
338  break;
339  }
340  value += c;
341  break;
342 
343  default:
344  throw logic_error("Invalid state while trying to parse logging command!");
345  break;
346  }
347  } //foreach char
348 
349  // Reject domain without value
350  if( ( (state == get_domain) && !domain.empty() ) ||
351  ( (state == get_value) && value.empty() )
352  ) {
353  throw domain_error(STR_OF << "No value given for domain " << domain);
354  }
355 
356  // Last one, if any
357  if(!domain.empty() && !value.empty()) {
358  setLevelFromStrings(domain, value);
359  }
360 
361  return true;
362 }
