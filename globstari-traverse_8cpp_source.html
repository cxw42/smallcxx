160 void
161 Traverser::worker()
162 {
163  while(!items_.empty()) {
164 
165  // Not auto& --- copy the item so we can pop_front() right away
166  const auto item = items_.front();
167  items_.pop_front();
168 
169  if(seen_.count(item.entry.canonPath)) {
170  LOG_F(TRACE, "already-seen %s --- skipping",
171  item.entry.canonPath.c_str());
172  continue;
173  }
174 
175  // TODO make sure this is in the right place
176  seen_.insert(item.entry.canonPath);
177 
178  if((maxDepth_ > 0) && (item.entry.depth > maxDepth_)) {
179  LOG_F(TRACE, "Skipping %s --- maxDepth exceeded",
180  item.entry.canonPath.c_str());
181  continue;
182  }
183 
184  // Check against the ignores we already have
185  if(item.ignores->contains(item.entry.canonPath)) {
186  LOG_F(TRACE, "ignored %s --- skipping",
187  item.entry.canonPath.c_str());
188  continue;
189  }
190 
191  // Is it a hit?
192  const auto match = needleMatcher_.check(item.entry.canonPath);
193 
194  LOG_F(TRACE, "%s %s", PathCheckResultNames[(int)match],
195  item.entry.canonPath.c_str());
196 
197  // Decide what to do
198  auto clientInstruction = GlobstariBase::ProcessStatus::Continue;
199 
200  if(match == PathCheckResult::Excluded) {
201  // Excluded => nothing more to do. Simple!
202  continue;
203 
204  } else if(match == PathCheckResult::Included) {
205  // Included => give it to the client. Also simple!
206  clientInstruction = client_->processEntry(item.entry);
207 
208  } else if(item.entry.ty == GlobstariBase::EntryType::Dir) {
209  // But directories not specifically included may contain
210  // files that are themselves included. Therefore,
211  // descend into directories if match == Unknown.
212  loadDir(item.entry, item.ignores);
213  continue;
214  }
215 
216  // Do what the client asked us to
217  switch(clientInstruction) {
218  case GlobstariBase::ProcessStatus::Continue:
219  if(item.entry.ty == GlobstariBase::EntryType::Dir) {
220  loadDir(item.entry, item.ignores);
221  }
222  break;
223 
224  case GlobstariBase::ProcessStatus::Skip:
225  // nothing to do
226  break;
227 
228 #if 0
229  // TODO implement this?
230  case GlobstariBase::ProcessStatus::Pop:
231  // TODO pop the deque until `dir` changes
232  break;
233 #endif
234 
235  case GlobstariBase::ProcessStatus::Stop:
236  throw StopTraversal();
237  break;
238 
239  default:
240  throw logic_error(STR_OF << "Unimplemented status value "
241  << (int)clientInstruction);
242  break;
243  } //switch(clientInstruction)
244 
245  } // while entries remain
