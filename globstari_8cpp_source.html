110 bool
111 Criteria::accepts(const smallcxx::glob::Path& str) const
112 {
113  MatchDataPtr matches(
114  pcre2_match_data_create_from_pattern(compiled_.get(), nullptr),
115  pcre2_match_data_free);
116 
117  int rc;
118  size_t *pcre_result;
119 
120  rc = pcre2_match(compiled_.get(), (PCRE2_SPTR8)str.c_str(), str.length(),
121  0, 0, matches.get(), nullptr);
122 
123  if (rc < 0) { /* failed to match */
124  if (rc == PCRE2_ERROR_NOMATCH) {
125  return false;
126  } else {
127  throw runtime_error(STR_OF
128  << "Failure while matching RE: code " << rc);
129  }
130  }
131 
132  pcre_result = pcre2_get_ovector_pointer(matches.get());
133 
134  // Did anything match?
135  if((pcre_result[1] == 0 ) ||
136  (pcre_result[0] == PCRE2_UNSET && pcre_result[1] == PCRE2_UNSET)) {
137  LOG_F(FIXME, "Zero-length successful match --- probably a bug! >>%s<<",
138  str.c_str());
139  return false;
140  }
141 
142  /* Check whether the numbers are in the desired range */
143  size_t i;
144  auto rangeit = ranges_.cbegin();
145  for(i = 1; rangeit != ranges_.cend(); ++i, ++rangeit) {
146 
147  const auto ofsBegin = pcre_result[2 * i];
148  const auto ofsEnd = pcre_result[2 * i + 1];
149 
150  // Didn't match this capturing group --- keep going
151  if(ofsBegin == PCRE2_UNSET && ofsEnd == PCRE2_UNSET) {
152  continue;
153  }
154 
155  const char *substring_start = str.data() + pcre_result[2 * i];
156  size_t substring_length = pcre_result[2 * i + 1] - pcre_result[2 * i];
157 
158  /* we don't consider 0 digits such as 010 as matched */
159  if (*substring_start == '0') {
160  return false;
161  }
162 
163  if(substring_length == 0) {
164  // XXX is this the right thing?
165  throw runtime_error(STR_OF << "Zero length substring match at"
166  << " index " << i);
167  }
168 
169  string num_string(substring_start, substring_length);
170  Int num = strtoll(num_string.c_str(), nullptr, 10);
171 
172  if (num < rangeit->first || num > rangeit->second) { /* not matched */
173  return false; // it has to match all of them
174  }
175  } // for each range
176 
177  return true;
178 }

186 static void
187 globToRegexSrc(const smallcxx::glob::Path& glob, string& src,
188  RangePairs& ranges)
189 {
190  // Places we need to force in a backslash.
191  // Invariant: the end of the string can't be used.
192  unordered_set<const char *> toBackslash;
193 
194  const char *c;
195  int brace_level = 0;
196  bool is_in_bracket = false;
197  int error_code;
198  size_t erroffset;
199  pcre2_code *reNumRaw;
200  int rc;
201  bool are_braces_paired = true;
202 
203  /* Determine whether curly braces are paired */
204  {
205  int left_count = 0;
206  int right_count = 0;
207  for (c = &glob[0]; *c; ++ c) {
208  if (*c == '\\' && *(c + 1) != '\0') {
209  ++ c;
210  continue;
211  }
212 
213  if (*c == '}') {
214  ++ right_count;
215  } else if (*c == '{') {
216  ++ left_count;
217  }
218 
219  if (right_count > left_count) {
220  are_braces_paired = false;
221  break;
222  }
223  }
224 
225  if (right_count != left_count) {
226  are_braces_paired = false;
227  }
228  }
229 
230  /* used to search for {num1..num2} case */
231  // TODO make this a singleton
232  reNumRaw = pcre2_compile((PCRE2_SPTR8)
233  "^\\{[\\+\\-]?\\d+\\.\\.[\\+\\-]?\\d+\\}$",
234  PCRE2_ZERO_TERMINATED, 0,
235  &error_code, &erroffset, nullptr);
236  if (!reNumRaw) { /* failed to compile */
237  throw std::runtime_error(STR_OF << "Could not create reNum: "
238  << error_code << " at offset " << erroffset);
239  }
240  RePtr pReNum(reNumRaw, pcre2_code_free);
241  reNumRaw = nullptr;
242 
243  for (c = &glob[0]; *c; ++ c) {
244 
245  // Force in backslashes
246  if(toBackslash.find(c) != toBackslash.end()) {
247  if (*c != '\0') {
248  src += '\\';
249  src += *c;
250  } else {
251  throw logic_error("toBackslash invariant violation");
252  }
253  continue;
254  }
255 
256  switch (*c) {
257  case '\\': /* also skip the next one */
258  if (*(c + 1) != '\0') {
259  src += *(c++);
260  src += *c;
261  } else {
262  src += "\\\\";
263  }
264 
265  break;
266  case '?':
267  src += "[^/]";
268  break;
269  case '*':
270  if (*(c + 1) == '*') { /* case of ** */
271  src += ".*";
272  ++ c;
273  } else { /* case of * */
274  src += "[^\\/]*";
275  }
276 
277  break;
278  case '[':
279  if (is_in_bracket) { /* inside brackets, we really mean bracket */
280  src += "\\[";
281  break;
282  }
283 
284  {
285  /* check whether we have slash within the bracket */
286  bool has_slash = false;
287  const char *cc;
288  for (cc = c; *cc && *cc != ']'; ++ cc) {
289  if (*cc == '\\' && *(cc + 1) != '\0') {
290  ++ cc;
291  continue;
292  }
293 
294  if (*cc == '/') {
295  has_slash = 1;
296  break;
297  }
298  }
299 
300  /* if we have slash in the brackets, just do it literally */
301  if (has_slash) {
302  const char *right_bracket = strchr(c, ']');
303 
304  if (!right_bracket) { /* The right bracket may not exist */
305  right_bracket = c + strlen(c);
306  }
307 
308  src += "\\";
309  copy(c, right_bracket, back_inserter(src));
310  if (*right_bracket) { /* right_bracket is a bracket, not \0 */
311  src += "\\]";
312  }
313  c = right_bracket;
314  if (!*c)
315  /* end of string, meaning that right_bracket is not a
316  * bracket. Then we go back one character to make the
317  * parsing end normally for the counter in the "for"
318  * loop. */
319  {
320  c -= 1;
321  }
322  break;
323  }
324  }
325 
326  is_in_bracket = 1;
327  if (*(c + 1) == '!') { /* case of [!...] */
328  src += "[^";
329  ++ c;
330  } else {
331  src += '[';
332  }
333 
334  break;
335 
336  case ']':
337  is_in_bracket = 0;
338  src += *c;
339  break;
340 
341  case '-':
342  if (is_in_bracket) { /* in brackets, - indicates range */
343  src += *c;
344  } else {
345  src += "\\-";
346  }
347 
348  break;
349  case '{':
350  if (!are_braces_paired) {
351  src += "\\{";
352  break;
353  }
354 
355  /* Check the case of {single}, where single can be empty */
356  {
357  const char *cc;
358  bool is_single = true;
359 
360  for (cc = c + 1; *cc != '\0' && *cc != '}'; ++ cc) {
361  if (*cc == '\\' && *(cc + 1) != '\0') {
362  ++ cc;
363  continue;
364  }
365 
366  if (*cc == ',') {
367  is_single = 0;
368  break;
369  }
370  } // for cc
371 
372  if (*cc == '\0') {
373  is_single = 0; // XXX is this unreachable?
374  }
375 
376  if (is_single) { /* escape the { and the corresponding } */
377  // A {} block with no commas in it
378  const char *double_dots;
379  IntPair intpair;
380 
381  pcre2_match_data *match_data = pcre2_match_data_create_from_pattern(
382  pReNum.get(), nullptr);
383 
384  /* Check the case of {num1..num2} */
385  // c points to the `{`. cc points to the `}`.
386  rc = pcre2_match(pReNum.get(),
387  (PCRE2_SPTR8)c, cc - c + 1, 0, 0, match_data,
388  nullptr);
389 
390  pcre2_match_data_free(match_data);
391 
392  if (rc < 0) { /* not {num1..num2} case */
393  src += "\\{";
394

396  // need to insert a \\ just before the rbrace.
397  toBackslash.insert(cc);
398 
399  break;
400  }
401 
402  /* Get the range */
403  double_dots = strstr(c, "..");
404  intpair.first = strtoll(c + 1, nullptr, 10);
405  intpair.second = strtoll(double_dots + 2, nullptr, 10);
406 
407  ranges.push_back(intpair);
408 
409  src += "([\\+\\-]?\\d+)";
410  c = cc;
411 
412  break;
413  } // endif is_single
414  } // end anonymous scope
415 
416  ++ brace_level;
417  src += "(?:";
418  break;
419 
420  case '}':
421  if (!are_braces_paired) {
422  src += "\\}";
423  break;
424  }
425 
426  -- brace_level;
427  src += ')';
428  break;
429 
430  case ',':
431  if (brace_level > 0) { /* , inside {...} */
432  src += '|';
433  } else {
434  src += "\\,";
435  }
436  break;
437 
438  case '/':
439  // /**/ case, match both single / and /anything/
440  if (!strncmp(c, "/**/", 4)) {
441  src += "(\\/|\\/.*\\/)";
442  c += 3;
443  } else {
444  src += "\\/";
445  }
446 
447  break;
448 
449  default:
450  if (!isalnum(*c)) {
451  src += '\\';
452  }
453 
454  src += *c;
455  } // switch(*c)
456  } // for c

505 void
506 GlobSetImpl::finalize()
507 {
508  // regex for all the globs that don't have numerical ranges
509  string nonRangeSrc("^(?:");
510  bool hasNonRange = false;
511 
512  for(const auto& glob : globs_) {
513  string reSrc;
514  RangePairs ranges;
515 
516  globToRegexSrc(glob, reSrc, ranges);
517 
518  if(ranges.empty()) {
519  // No numeric groups --- can be combined with all other globs
520  // having no numeric groups
521 
522  hasNonRange = true;
523 
524  // atomic group --- each individual branch of the glob
525  // will either match or it won't.
526  nonRangeSrc += "(?>";
527  nonRangeSrc += reSrc;
528  nonRangeSrc += ")|";
529 
530  } else {
531  // Has numeric groups --- must be evaluated on its own.
532  // This is because otherwise all numeric values textually match
533  // at the first numeric-group placeholder in a regex, so we
534  // can't get past the first option. E.g.,
535  //
536  // Globs: {1..10}, {101..110}
537  // Regex (simplified): /^(?:(\d+)|(\d+))$/
538  // The match never gets to $2, because digits are caught by $1.
539 
540  reSrc = "^(?>" + reSrc + ")$";
541  criteria_.emplace_back(reSrc, ranges);
542  }
543 
544  } // foreach glob
545 
546  if(hasNonRange) {
547  nonRangeSrc += "(*FAIL)"; // need something after the last `|`
548  nonRangeSrc += ")$";
549 
550  // Check the non-range ones first since we can dispose of a bunch
551  // in a single check. Also, my experience is that globs without
552  // numeric ranges will be much more common than globs with
553  // numeric ranges.
554  criteria_.emplace_front(nonRangeSrc, RangePairs{});
555  }
556 
557  finalized_ = true;
558 } // GlobSetImpl::finalize()
